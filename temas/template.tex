\noindent Este documento descreve um protocolo de \textbf{mini-cut estruturado de 30 dias}
para recomposição corporal, fundamentado em evidências científicas revisadas por pares.
Não substitui aconselhamento médico ou nutricional individual.

\vspace{0.8em}
\noindent\textbf{Organização:}
\begin{itemize}[itemsep=2pt, topsep=4pt]
  \item \textbf{Capítulos 1–6, 10} — Núcleo do protocolo (execução diária)
  \item \textbf{Capítulos 7, 9, 11, 15} — Segurança e monitoramento
  \item \textbf{Capítulos 12, 14, 16–17, 20–21} — Desempenho e recuperação
  \item \textbf{Capítulos 8, 13, 18} — Contextos especiais
  \item \textbf{Capítulos 19, 22–24} — Adaptações populacionais
  \item \textbf{Glossário} — 142 termos, 13 especialidades profissionais
  \item \textbf{Referências} — 142 fontes científicas (\textit{[web:1]} a \textit{[web:142]})
\end{itemize}

\vspace{0.8em}
\noindent\textbf{Público-alvo:} adultos saudáveis de 18–60 anos. Populações especiais
(mulheres, idosos ≥ 65 anos, menopausa, crianças/adolescentes) devem consultar os
capítulos específicos antes de iniciar.

\vspace{0.8em}
\noindent\textbf{Convenções tipográficas:}
\begin{itemize}[itemsep=2pt, topsep=4pt]
  \item Referências científicas são indicadas por \textsuperscript{\small\color{AccentFg}[web:N]}
